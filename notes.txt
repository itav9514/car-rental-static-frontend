
server {

    server_name abhirajtoys.com www.abhirajtoys.com;

    server_tokens off;

    # Root for frontend SPA (React/Vue/etc.)
    root /var/www/abhiraj-toys-static-frontend;
    index index.html;

    # Clean up .html extensions and index.html
    rewrite ^/(.*)\.html(\?.*)?$ /$1$2 permanent;
    rewrite ^(.*)/index\.html$ $1 permanent;
    rewrite ^/index\.html$ / permanent;

    location / {
        try_files $uri $uri/ $uri.html /index.html;
    }

    # Cache static assets for frontend
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot)$ {
        expires 30d;
        access_log off;
        add_header Cache-Control "public";
    }



}
